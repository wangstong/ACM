#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
const int mod = 1e9+7;

int main() {
    ll n, ans;
    scanf("%lld", &n);
    ans = ((n%mod)*(n%mod)) % mod;
    ans = ans + ((n%mod)*((n-1) % mod)) % mod;
    ll t = sqrt(n);

    for (ll ac = 2; ac <= t; ac++) {
        for (ll c = ac*ac, b = 2; c <= n; c *= ac, b++) {
            for (ll d = 1; d < b; d++) {
                if (__gcd(d,b) == 1) {
                    ans = (ans + ((n / b) * 2) % mod) % mod;  // *2代表ab和cd交换所以*2
                }
            }
        }
    }
    printf("%lld\n", ans);
    return 0;
}
