#include<bits/stdc++.h>

using namespace std;

int main() {
    int n;
    cin >> n;
    if(n == 1)
        puts("2 3 4");
    else if (n==2) {
        puts("2 4 5");
        puts("3 6 7");
    }
    else {
        int k = n+1;
        if (n % 2 == 1) {
            printf("%d %d %d\n", n+1, 2*n+1, 3*n+1);
            k--;
        }
        int k1 = n+1+(k/2), k2 = n+1+(k/2+1);
        int k3 = 2*n+1+(k/2), k4 = 2*n+1+(k/2+1);
        for(int i = 2; i <= k; i++){
            if (i % 2 == 0) {
                printf("%d %d %d\n", i, k1--, k2++);
            }
            else {
                printf("%d %d %d\n", i, k3--, k4++);
            }
        }
    }
    return 0;
}
